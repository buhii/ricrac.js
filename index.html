jquery.ricrac.js

You can create a simple *1D* button array on canvas. Please click, touch or drag the below box:

pos:

data:

Requirements

  • jQuery
  • Canvas Element

Demos

  • With border(resizable)
  • Without border, 1000+ buttons(resizable)
  • Initial data, read-only
  • Initial data, read-only(resizable)

With border(resizable)

$("#with-border").ricrac({
    dataRange: [1, 35],
    gridColumn: 6,
    borderWidth: 8,
    resizable: true,
    borderColor: "#ccf",
    colorOn: "#5c7",
    colorOff: "#fff",
    gridSize: 32,   // px
});

Without border, 1000+ buttons(resizable)

$("#without").ricrac({
    dataRange: [7, 1550],
    gridColumn: 50,
    borderWidth: 0,
    resizable: true,
    gridSize: 10,
});

Initial data, read-only

$("#read-only").ricrac({
    readonly: true,
    dataRange: [1, 25],
    initData: "0101001010000001000101110".split("").map(function(v) {
      return parseInt(v);
    }),
    gridColumn: 5,
    borderWidth: 0,
    gridSize: 16,
    colorOn: "#a75",
    colorOff: "#fff",
});

Initial data, read-only(resizable)

var gcColumn = $("#gc-column").ricrac({
    initData: "01010101010101010101010101".split("").map(function(v) {
        return parseInt(v);
    }),
    readonly: true,
    resizable: true,
    dataRange: [1, 25],
    gridColumn: 5,
    borderWidth: 2,
    gridSize: 32,
});
